main :: IO ()
main = do
    putStrLn "Qual e o seu nome?"
    nome <- getLine -- Le a entrada do teclado
    
    putStrLn "Em que ano voce nasceu?"
    anoStr <- getLine
    
    let anoNascimento = read anoStr :: Int
        anoAtual = 2026
        idade = anoAtual - anoNascimento
    
    putStrLn ("Ola, " ++ nome ++ "!")
    
    -- Estrutura condicional para verificar a maioridade
    if idade >= 18
        then putStrLn "Voce ja e maior de idade."
        else putStrLn "Voce e menor de idade."
